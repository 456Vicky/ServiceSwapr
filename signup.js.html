signup.js 
1    import { auth, createUserWithEmailAndPassword } from './firebaseconfig.js';
2    
3    document.addEventListener('DOMContentLoaded', () => {
4        const signupForm = document.getElementById('signup-form');
5        signupForm.addEventListener('submit', async (e) => {
6            e.preventDefault();
7    
8            const email = document.getElementById('wf-sign-up-email').value;
9            const password = document.getElementById('wf-sign-up-password').value;
10   
11           try {
12               const userCredential = await createUserWithEmailAndPassword(auth, email, password);
13               const user = userCredential.user;
14               console.log('User signed up:', user);
15               alert('Signup successful!');
16               window.location.href = 'index.html'; // Redirect after successful signup
17           } catch (error) {
18               const errorCode = error.code;
19               const errorMessage = error.message;
20               console.error('Error signing up:', errorCode, errorMessage);
21               alert(`Error: ${errorMessage}`);
22           }
23       });
24   });
